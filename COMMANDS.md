# GIT COMMANDS

## for status
git status

## for repository clone & install

git clone "github repository url"
# inside the cloned repository folder 
npm i 


## GIT PULL

git pull origin main --allow-unrelated-histories
git add .
git commit -m "merge histories"
git branch -M main 
git push origin main

## GIT PUSH

git status
git add . 
# or
git add <file>
git commit -m "merge histories"
git push origin main
